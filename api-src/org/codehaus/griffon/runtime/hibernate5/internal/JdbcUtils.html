031 /**
032  * <p>Generic utility methods for working with JDBC.</p>
033  * <p>Contains code from Spring's {@code org.springframework.jdbc.support.JdbcUtils}.</p>
034  * <p/>
035  * Original author Thomas Risberg<br/>
036  * Original author Juergen Hoeller<br/>
037  *
038  * @author Andres Almiray
039  */
040 public abstract class JdbcUtils {
041     private static final Logger LOG = LoggerFactory.getLogger(HibernateConfigurationHelper.class);
042 
043     /**
044      * Close the given JDBC Connection and ignore any thrown exception.
045      * This is useful for typical finally blocks in manual JDBC code.
046      *
047      * @param con the JDBC Connection to close (may be <code>null</code>)
048      */
049     public static void closeConnection(Connection con) {
050         if (con != null) {
051             try {
052                 con.close();
053             } catch (SQLException ex) {
054                 if (LOG.isTraceEnabled()) {
055                     LOG.debug("Could not close JDBC Connection", ex);
056                 }
057             } catch (Throwable ex) {
058                 // We don't trust the JDBC driver: It might throw RuntimeException or Error.
059                 if (LOG.isTraceEnabled()) {
060                     LOG.debug("Unexpected exception on closing JDBC Connection", ex);
061                 }
062             }
063         }
064     }
065 
066     /**
067      * Close the given JDBC Statement and ignore any thrown exception.
068      * This is useful for typical finally blocks in manual JDBC code.
069      *
070      * @param stmt the JDBC Statement to close (may be <code>null</code>)
071      */
072     public static void closeStatement(Statement stmt) {
073         if (stmt != null) {
074             try {
075                 stmt.close();
076             } catch (SQLException ex) {
077                 if (LOG.isTraceEnabled()) {
078                     LOG.trace("Could not close JDBC Statement", ex);
079                 }
080             } catch (Throwable ex) {
081                 if (LOG.isTraceEnabled()) {
082                     // We don't trust the JDBC driver: It might throw RuntimeException or Error.
083                     LOG.trace("Unexpected exception on closing JDBC Statement", ex);
084                 }
085             }
086         }
087     }
088 
089     /**
090      * Close the given JDBC ResultSet and ignore any thrown exception.
091      * This is useful for typical finally blocks in manual JDBC code.
092      *
093      * @param rs the JDBC ResultSet to close (may be <code>null</code>)
094      */
095     public static void closeResultSet(ResultSet rs) {
096         if (rs != null) {
097             try {
098                 rs.close();
099             } catch (SQLException ex) {
100                 if (LOG.isTraceEnabled()) {
101                     LOG.trace("Could not close JDBC ResultSet", ex);
102                 }
103             } catch (Throwable ex) {
104                 // We don't trust the JDBC driver: It might throw RuntimeException or Error.
105                 if (LOG.isTraceEnabled()) {
106                     LOG.trace("Unexpected exception on closing JDBC ResultSet", ex);
107                 }
108             }
109         }
110     }

112     /**
113      * Extract database meta data via the given DatabaseMetaDataCallback.
114      * <p>This method will open a connection to the database and retrieve the database metadata.
115      * Since this method is called before the exception translation feature is configured for
116      * a datasource, this method can not rely on the SQLException translation functionality.
117      * <p>Any exceptions will be wrapped in a MetaDataAccessException. This is a checked exception
118      * and any calling code should catch and handle this exception. You can just log the
119      * error and hope for the best, but there is probably a more serious error that will
120      * reappear when you try to access the database again.
121      *
122      * @param dataSource the DataSource to extract metadata for
123      * @param action     callback that will do the actual work
124      * @return object containing the extracted information, as returned by
125      * the DatabaseMetaDataCallback's <code>processMetaData</code> method
126      * @throws MetaDataAccessException if meta data access failed
127      */
128     public static Object extractDatabaseMetaData(DataSource dataSource, DatabaseMetaDataCallback action)
129         throws MetaDataAccessException {
130 
131         Connection con = null;
132         try {
133             con = dataSource.getConnection();
134             if (con == null) {
135                 // should only happen in test environments
136                 throw new MetaDataAccessException("Connection returned by DataSource [" + dataSource + "] was null");
137             }
138             DatabaseMetaData metaData = con.getMetaData();
139             if (metaData == null) {
140                 // should only happen in test environments
141                 throw new MetaDataAccessException("DatabaseMetaData returned by Connection [" + con + "] was null");
142             }
143             return action.processMetaData(metaData);
144         } catch (SQLException ex) {
145             throw new MetaDataAccessException("Error while extracting DatabaseMetaData", ex);
146         } catch (AbstractMethodError err) {
147             throw new MetaDataAccessException(
148                 "JDBC DatabaseMetaData method not implemented by JDBC driver - upgrade your driver", err);
149         } finally {
150             closeConnection(con);
151         }
152     }
153 
154     /**
155      * Call the specified method on DatabaseMetaData for the given DataSource,
156      * and extract the invocation result.
157      *
158      * @param dataSource         the DataSource to extract meta data for
159      * @param metaDataMethodName the name of the DatabaseMetaData method to call
160      * @return the object returned by the specified DatabaseMetaData method
161      * @throws MetaDataAccessException if we couldn't access the DatabaseMetaData
162      *                                 or failed to invoke the specified method
163      * @see java.sql.DatabaseMetaData
164      */
165     public static Object extractDatabaseMetaData(DataSource dataSource, final String metaDataMethodName)
166         throws MetaDataAccessException {
167         return extractDatabaseMetaData(dataSource,
168             new DatabaseMetaDataCallback() {
169                 public Object processMetaData(DatabaseMetaData dbmd) throws SQLException, MetaDataAccessException {
170                     try {
171                         Method method = DatabaseMetaData.class.getMethod(metaDataMethodName, (Class[]) null);
172                         return method.invoke(dbmd, (Object[]) null);
173                     } catch (NoSuchMethodException ex) {
174                         throw new MetaDataAccessException("No method named '" + metaDataMethodName +
175                             "' found on DatabaseMetaData instance [" + dbmd + "]", ex);
176                     } catch (IllegalAccessException ex) {
177                         throw new MetaDataAccessException(
178                             "Could not access DatabaseMetaData method '" + metaDataMethodName + "'", ex);
179                     } catch (InvocationTargetException ex) {
180                         if (ex.getTargetException() instanceof SQLException) {
181                             throw (SQLException) ex.getTargetException();
182                         }
183                         throw new MetaDataAccessException(
184                             "Invocation of DatabaseMetaData method '" + metaDataMethodName + "' failed", ex);
185                     }
186                 }
187             });
188     }
